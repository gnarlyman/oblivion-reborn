scn DLCBattlehornCastleScript

float fQuestDelayTime
short TotalBought				; counts number of items purchased

short Sparring					; set to 1 while player is sparring with trainer

short TrophyCount				; counts number of trophies created

ref FollowerRef					; set to knight who is currently following player
short Following					; 0 = no one following; 1 = following; 2 = waiting
short FollowerClear			; set to 1 to clear follower variables

short Following01				; set to 1 when knight is following player, 2 when waiting
short Following02				; set to 1 when knight is following player, 2 when waiting
short Following03				; set to 1 when knight is following player, 2 when waiting
short Following04				; set to 1 when knight is following player, 2 when waiting
short Following05				; set to 1 when knight captain is following player, 2 when waiting

short MaidFollow				; set to 1 when maid is following

short	AttackersDead			; tracks how many attackers have been killed

begin gamemode

;kickoff block

if ( GetStage DLCBattlehornCastle <= 0 ) && ( Player.IsInCombat == 0 ) && ( GetStage CharacterGen >= 88 ) && ( Player.GetinCell ImperialDungeon03 == 0 )
	SetStage DLCBattlehornCastle 10
endif

;arrival trigger

if GetStage DLCBattlehornCastle == 10
	if ( Player.GetDistance BattlehornCastleGateREF <= 5000 )
		SetStage DLCBattlehornCastle 18
	endif
endif

if GetStage DLCBattlehornCastle == 18
	if ( Player.GetDistance BattlehornCastleGateREF <= 2000 )
		SetStage DLCBattlehornCastle 20
	endif
endif


; attackers all dead
if getStage DLCBattlehornCastle < 28
	if AttackersDead >= 4
		setstage DLCBattlehornCastle 28
	endif
endif

; adds forge receipt to vendor after you buy one thing
if TotalBought > 0 && GetStageDone DLCBattlehornCastle 40 == 0
	if player.getincell BattlehornCastle == 1
		setstage DLCBattlehornCastle 40
		DLCBhornKnightRef.Evp
	endif
endif

;***Handle clearing of follower variables***

	if ( FollowerClear == 1 )
		set Following01 to 0
		set Following02 to 0
		set Following03 to 0
		set Following04 to 0
		set Following05 to 0
		set Following to 0
		set FollowerClear to 0
	endif


end

scriptname DLC06ThievesDenQuestScript

short	SkeletonDeathCount
short 	DunbarrowUpgradeCount
short 	StolenGoldHold
short 	StolenGoldHoldProduct
short 	GiveLoot
short	PiratesReady
short	PiratesGo
short	PiratesOut
short 	LastPickup
short 	doOnce

float 	fQuestDelayTime

BEGIN GameMode

if ( getStage DLC06ThievesDen == 0 )  && ( player.IsInCombat == 0 ) && ( getStage CharacterGen >= 88 ) 
	setstage DLC06ThievesDen 10
	set DunbarrowUpgradeCount to 0
	if ( GetStageDone TG00FindThievesGuild 30 == 1 )
		setStage TG00FindThievesGuild 20 
	endif
endif

if ( getStage DLC06ThievesDen == 10 )
	set fQuestDelayTime to .1
	if ( Player.GetDistance DLCDunbarrowMapMarker <= 4000 )
		SetStage DLC06ThievesDen 11
		set fQuestDelayTime to 0
	endif
endif

if ( getStage DLC06ThievesDen >= 11 ) && ( getstage DLC06ThievesDen < 100 )

	if ( getStage DLC06ThievesDen >= 10 ) && ( getStage DLC06ThievesDen <= 11 )   && ( player.getInCell DLDunBarrowHaven == 1 )
		setstage DLC06ThievesDen 15
	endif

	if ( GetStage DLC06ThievesDen == 15 ) && ( SkeletonDeathCount >= 7 )
		setstage DLC06ThievesDen 18
	endif

	if ( GetStage DLC06ThievesDen >= 18 )
		if ( player.getInCell DLDunBarrowHaven == 0 )
			if ( player.getInCell DLDunBarrowHaven02 == 0 ) 
				if ( player.getInCell DLDunBarrowHaven03 == 0 )
					DLC06PirateSkeleton01.disable 
					DLC06PirateSkeleton02.disable 
					DLC06PirateSkeleton03.disable 
					DLC06PirateSkeleton04.disable 
					DLC06PirateSkeleton05.disable 
					DLC06PirateSkeleton06.disable 
					DLC06PirateCaptain.disable 
				endif
			endif
		endif
	endif

	if ( DunbarrowUpgradeCount >= 1 ) && ( DoOnce == 0 )
		DLClutterParent.Enable
		set DoOnce to 1
	endif

	if ( DunbarrowUpgradeCount >= 2 ) && ( DoOnce == 1 )
		DLCPirate01Ref.enable
		Message "Due to the prestige of your crew, the pirate Zedrick Green has joined."
		set DoOnce to 2
	endif

	if ( DunbarrowUpgradeCount >= 4 )  && ( DoOnce == 2 )
		DLCPirate02Ref.enable
		Message "Due to the prestige of your crew, the pirate Scurvy John Hoff has joined."
		set DoOnce to 3
	endif

	if ( DunbarrowUpgradeCount >= 6 )  && ( DoOnce == 3 )
		DLCPirate03Ref.enable
		Message "Due to the prestige of your crew, the pirate Yinz'r has joined."
		setstage DLC06ThievesDen 100
		Set DunbarrowUpgradeCount to 0
		set DoOnce to 4
	endif

endif

END

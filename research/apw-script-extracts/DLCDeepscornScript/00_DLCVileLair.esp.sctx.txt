scn DLCDeepscornScript

float fQuestDelayTime
short Arrived
short Book
short Doonce
short Doonce1
short Doonce2
short Doonce3
short Doonce4
short Doonce5
short Doonce6
short Doonce7
short TotalCount

begin gamemode

;kickoff block

if ( GetStage DLCDeepscorn <= 0 ) && ( Player.IsInCombat == 0 ) && ( GetStage CharacterGen >= 88 ) && ( Player.GetinCell ImperialDungeon03 == 0 )
	SetStage DLCDeepscorn 10
endif

;arrival trigger

if ( Player.GetDistance DeepscornHollowMapMarker <= 500 ) && ( GetStage DLCDeepscorn == 10 ) && ( Arrived == 0 )
	SetStage DLCDeepscorn 20
	Set Arrived to 1
endif

;book trigger

if ( Player.GetDistance DeepscornHollowJournalRef <= 200 ) && ( GetStage DLCDeepscorn >= 20 ) && ( Book == 0 )
	SetStage DLCDeepscorn 30
	Set Book to 1
endif

;These are the variable triggers that are set when a voucher is acquired

if ( Player.GetItemCount DeepscornBedroomReceipt == 1 ) && ( Doonce == 0 )
	LairVileBedroomMarker.Enable
	set TotalCount to ( TotalCount + 1 )
	set Doonce to 1
endif

if ( Player.GetItemCount DeepscornCellReceipt == 1 ) && ( Doonce1 == 0 )
	LairVilePrisonMarker.Enable
	set TotalCount to ( TotalCount + 1 )
	set Doonce1 to 1
endif

if ( Player.GetItemCount DeepscornDiningReceipt == 1 ) && ( Doonce2 == 0 )
	LairVileDiningArea.Enable
	set TotalCount to ( TotalCount + 1 )
	set Doonce2 to 1
endif

if ( Player.GetItemCount DeepscornMinionReceipt == 1 ) && ( Doonce3 == 0 )
	LairVileBoxingMarker.Enable
	LairVileMinionRef.Enable
	set TotalCount to ( TotalCount + 1 )
	set Doonce3 to 1
endif

if ( Player.GetItemCount DeepscornStorageReceipt == 1 ) && ( Doonce4 == 0 )
	LairVileMinionStockMarker.Enable
	set TotalCount to ( TotalCount + 1 )
	set Doonce4 to 1
endif

if ( Player.GetItemCount DeepscornStudyReceipt == 1 ) && ( Doonce5 == 0 )
	LairVileStudy.Enable
	set TotalCount to ( TotalCount + 1 )
	set Doonce5 to 1
endif

if ( Player.GetItemCount DeepscornVenomgrowthReceipt == 1 ) && ( Doonce6 == 0 )
	LairVileTendingMarker.Enable
	set TotalCount to ( TotalCount + 1 )
	set Doonce6 to 1
endif

if ( Player.GetItemCount IchorofSithis == 1 ) && ( Doonce7 == 0 )
	set TotalCount to ( TotalCount + 1 )
	set Doonce7 to 1
endif

;Ends Quest if All Bought

if ( TotalCount == 8 )
	LairVileArmoireRef.Enable
	setstage DLCDeepscorn 50
	set fQuestDelayTime to 0
endif

end
